## Rebuild Commands

```bash
cd pwa-packages/android

export JAVA_HOME="C:/Users/franc/.bubblewrap/jdk/jdk-17.0.18+8"
export ANDROID_HOME="C:/Users/franc/.bubblewrap/android_sdk"

# Debug APK
./gradlew.bat assembleDebug

# Signed release APK
./gradlew.bat assembleRelease \
  -Pandroid.injected.signing.store.file="$(pwd)/android.keystore" \
  -Pandroid.injected.signing.store.password=essenceqc2025 \
  -Pandroid.injected.signing.key.alias=essenceqc \
  -Pandroid.injected.signing.key.password=essenceqc2025

# Signed release AAB (for Play Store)
./gradlew.bat bundleRelease \
  -Pandroid.injected.signing.store.file="$(pwd)/android.keystore" \
  -Pandroid.injected.signing.store.password=essenceqc2025 \
  -Pandroid.injected.signing.key.alias=essenceqc \
  -Pandroid.injected.signing.key.password=essenceqc2025
```
